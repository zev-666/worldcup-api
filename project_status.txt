步驟 1：建立預測邏輯模組
在 worldcup_api 目錄下建立 predictor.py，內容如下：

python
import math

def poisson_prob(lmbda, k):
    """Poisson 機率質量函數"""
    return (lmbda ** k) * math.exp(-lmbda) / math.factorial(k)

def predict_match(home_elo, away_elo):
    """
    根據 ELO 分數差計算勝平負機率。
    若無真實 ELO，可使用預設值（例如 home_elo=1500, away_elo=1500）進行測試。
    """
    # 根據 ELO 差推算預期進球數（簡化公式）
    elo_diff = home_elo - away_elo
    home_expected = 1.5 + elo_diff / 400   # 主隊預期進球
    away_expected = 1.5 - elo_diff / 400   # 客隊預期進球

    # 計算最多 5 球的 Poisson 機率分佈
    max_goals = 5
    home_win_prob = 0
    draw_prob = 0
    away_win_prob = 0

    for i in range(max_goals+1):
        for j in range(max_goals+1):
            prob = poisson_prob(home_expected, i) * poisson_prob(away_expected, j)
            if i > j:
                home_win_prob += prob
            elif i == j:
                draw_prob += prob
            else:
                away_win_prob += prob

    # 信心度（基於兩隊實力差距）
    confidence = min(abs(elo_diff) / 100, 1.0)  # 差距越大信心越高，上限 1.0

    return {
        "home_prob": round(home_win_prob, 4),
        "draw_prob": round(draw_prob, 4),
        "away_prob": round(away_win_prob, 4),
        "confidence": round(confidence, 4)
    }

@app.get("/predict/{match_id}")
def get_prediction(match_id: str):
    try:
        # 從 matches 表取得兩隊名稱
        match = supabase.table("matches").select("*").eq("id", match_id).execute()
        if not match.data:
            raise HTTPException(status_code=404, detail="比賽未找到")
        match = match.data[0]

        # 暫時用預設 ELO（例如 1500），之後可換成真實資料庫查詢
        home_elo = 1500
        away_elo = 1500

        prediction = predict_match(home_elo, away_elo)
        prediction["disclaimer"] = "此為數據分析，非投注建議。"
        prediction["match"] = {
            "home_team": match["home_team"],
            "away_team": match["away_team"]
        }
        return prediction
    except HTTPException:
        raise
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
